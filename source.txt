# Exemplo de Programa 
# Comentarios começam usando '#' até '\n'

print "Digite um numero: ";				# print e read são keywords de io
read x;							# comandos terminam com ';'
							# ainda não sei como armazenar strings ???

	print "Este numero é positivo\n";		# blocos de comandos começam com '{' e terminam com '}'

	print "Este numero é negativo\n";
}

i = 0;

	print "dentro do loop\n";
}
